rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ---- SAVES: cada jogador só lê e escreve o próprio save ----
    match /saves/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }

    // ---- SESSIONS: trava de dispositivo, só o dono acessa ----
    match /sessions/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }

    // ---- LEADERBOARD: todos leem; cada um só escreve a própria linha ----
    match /leaderboard/{uid} {
      allow read: if true;
      allow write: if request.auth != null
                   && request.auth.uid == uid
                   && request.resource.data.lvl is int
                   && request.resource.data.lvl >= 1
                   && request.resource.data.lvl <= 999
                   && request.resource.data.depth is int
                   && request.resource.data.depth >= 0
                   && request.resource.data.name is string
                   && request.resource.data.name.size() <= 20;
    }

    // ---- Tudo o mais fica bloqueado ----
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
